import React from "react";
import {
	BarChart,
	Bar,
	XAxis,
	YAxis,
	CartesianGrid,
	Tooltip,
	Legend,
	ResponsiveContainer,
	
} from "recharts";


/**
 * Component that displays th barChart with activities called in the dashboard page
 * @component
 */

const CustomTooltip = ({ active, payload }) => {
	if (active && payload && payload.length) {
		return (
			<div className="custom-tooltip-barchart">
				<p className="custom-tooltip-barchart-label">{`${payload[0]?.value} kg`}</p>
				<p className="custom-tooltip-barchart-label">{`${payload[1]?.value} KCal`}</p>
			</div>
		);
	}

	return null;
};
